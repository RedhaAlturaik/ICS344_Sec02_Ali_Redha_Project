const { LambdaClient, InvokeCommand } = require("@aws-sdk/client-lambda");
const { CognitoIdentityProviderClient, AdminGetUserCommand } = require("@aws-sdk/client-cognito-identity-provider");
const jose = require('node-jose');

exports.handler = (event, context, callback) => {

    try {
        // ✅ SAFE parsing (FIXED)
        const req = JSON.parse(event.body);
        const headers = event.headers || {};

        const auth_header = headers.Authorization || headers.authorization;
        if (!auth_header) {
            return callback(null, {
                statusCode: 401,
                headers: { "Access-Control-Allow-Origin": "*" },
                body: JSON.stringify({ status: "err", msg: "Missing Authorization" })
            });
        }

        const token_sections = auth_header.split('.');
        const auth_data = jose.util.base64url.decode(token_sections[1]);
        const token = JSON.parse(auth_data);
        const user = token.username;

        let isAdmin = false;

        const params = {
            UserPoolId: process.env.userpoolid,
            Username: user
        };

        const cognitoidentityserviceprovider = new CognitoIdentityProviderClient();
        const command = new AdminGetUserCommand(params);

        cognitoidentityserviceprovider.send(command).then((userData) => {

            // ✅ FIX: normalize boolean
            const attrs = userData.UserAttributes || [];
            for (let i = 0; i < attrs.length; i++) {
                if (attrs[i].Name === "custom:is_admin") {
                    const val = attrs[i].Value;
                    isAdmin = (val === true || val === "true");
                    break;
                }
            }

            let action = req.action;
            let isOk = true;
            let payload = {};
            let functionName = "";

            switch (action) {

                case "new":
                    payload = { user: user, cartId: req["cart-id"], items: req["items"] };
                    functionName = "DVSA-ORDER-NEW";
                    break;

                case "update":
                    // 🔒 FIX: enforce access control
                    if (!isAdmin) {
                        return callback(null, {
                            statusCode: 403,
                            headers: { "Access-Control-Allow-Origin": "*" },
                            body: JSON.stringify({
                                status: "err",

                            })
                        });

                        user: user,
                        orderId: req["order-id"],
                        items: req["items"]
                    };
                    functionName = "DVSA-ORDER-UPDATE";

                    functionName = "DVSA-ORDER-CANCEL";
                    break;

                case "get":
                    payload = {
                        user: user,
                        orderId: req["order-id"],
                        isAdmin: String(isAdmin) // ✅ FIX for Python bug
                    };
                    functionName = "DVSA-ORDER-GET";
                    break;

                    payload = { user: user };
                    functionName = "DVSA-ORDER-ORDERS";

                    functionName = "DVSA-USER-ACCOUNT";

                        user: user,

                        billing: req["data"]
